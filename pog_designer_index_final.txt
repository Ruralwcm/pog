  document.getElementById('toastContainer').appendChild(el);
  setTimeout(() => {
    el.style.animation = 'toastOut 0.3s ease forwards';
    setTimeout(() => el.remove(), 300);
  }, duration);
}

// ============================================================
// API HELPERS
// ============================================================
async function apiGet(params) {
  const url = new URL(API_URL);
  Object.entries(params).forEach(([k,v]) => url.searchParams.set(k, v));
  const res = await fetch(url.toString());
  if (!res.ok) throw new Error('HTTP ' + res.status);
  return res.json();
}

async function apiPost(body) {
  const res = await fetch(API_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
    redirect: 'follow'
  });
  if (!res.ok) throw new Error('HTTP ' + res.status);
  return res.json();
}

async function handleLogin() {
  const username = document.getElementById('loginUsername').value.trim();
  const password = document.getElementById('loginPassword').value.trim();
  if (!username || !password) {
    showToast('Vui lòng nhập đầy đủ thông tin', 'warning');
    return;
  }
  showLoading('Đang đăng nhập...');
  try {
    const data = await apiGet({ action: 'login', username, password });
    if (data.success) {
      currentUser = { username: data.username, role: data.role };
      localStorage.setItem('pogUser', JSON.stringify(currentUser));
      afterLogin();
    } else {
      showToast(data.error || 'Sai tên đăng nhập hoặc mật khẩu', 'error');
    }
  } catch(e) {
    showToast('Lỗi kết nối API. Kiểm tra lại URL.', 'error');
    console.error(e);
  } finally {
    hideLoading();
  }
}

async function loadProductsData() {
  if (allProducts.length > 0) return; // already loaded
  try {
    const data = await apiGet({ action: 'getProducts' });
    if (Array.isArray(data)) {
      allProducts = data;
      populateProductFilters();
      populateLibraryFilters();
    }
  } catch(e) {
    console.error('Failed to load products:', e);
  }
}

function loadProducts() {
  if (allProducts.length > 0) {
    renderProductGrid(allProducts);
    return;
  }
  showLoading('Đang tải sản phẩm...');
  apiGet({ action: 'getProducts' })
    .then(data => {
      if (Array.isArray(data)) {
        allProducts = data;
        populateProductFilters();
        populateLibraryFilters();
        filterProducts();
      }
    })
    .catch(e => showToast('Lỗi tải sản phẩm: ' + e.message, 'error'))
    .finally(() => hideLoading());
}

function populateProductFilters() {
  const danhMucs = [...new Set(allProducts.map(p => p.danhMuc).filter(Boolean))].sort();
  const sel = document.getElementById('filterDanhMuc');
  sel.innerHTML = '<option value="">-- Tất cả Danh mục --</option>';
  danhMucs.forEach(d => {
    const o = document.createElement('option');
    o.value = d; o.textContent = d;
    sel.appendChild(o);
  });
  filterProducts();
}

function filterProducts() {
  const danhMuc = document.getElementById('filterDanhMuc').value;
  const mch3 = document.getElementById('filterMCH3').value;
  const search = document.getElementById('searchProduct').value.toLowerCase().trim();

  // Update MCH3 dropdown based on danh mục
  const mch3Sel = document.getElementById('filterMCH3');
  const mch3s = [...new Set(allProducts
    .filter(p => !danhMuc || p.danhMuc === danhMuc)
    .map(p => p.mch3)
    .filter(Boolean)
  )].sort();

  const currentMCH3 = mch3Sel.value;
  mch3Sel.innerHTML = '<option value="">-- Tất cả MCH3 --</option>';
  mch3s.forEach(m => {
    const o = document.createElement('option');
    o.value = m; o.textContent = m;
    if (m === currentMCH3) o.selected = true;
    mch3Sel.appendChild(o);
  });

  filteredProducts = allProducts.filter(p => {
    if (danhMuc && p.danhMuc !== danhMuc) return false;
    if (mch3 && p.mch3 !== mch3) return false;
    if (search) {
      const hay = (p.tenSP + ' ' + p.maSP + ' ' + p.mch3).toLowerCase();
      if (!hay.includes(search)) return false;
    }
    return true;
  });

  document.getElementById('resultsCount').textContent = filteredProducts.length + ' sản phẩm';
  renderProductGrid(filteredProducts);
}

// ============================================================
// SVG SHELF GENERATOR
// ============================================================
function generateShelfSVG(shelf, svgW, svgH) {
  const { soTang, rong, caoTang, loai } = shelf;
  const ratio = rong / (soTang * caoTang);
  const padding = 8;
  const innerW = svgW - padding * 2;
  const innerH = svgH - padding * 2;

  let headerH = 0;
  const hasHeader = loai === 'tu';

  if (hasHeader) {
    headerH = Math.round(innerH * 0.22);
  }

  const bodyH = innerH - headerH;
  const tangH = bodyH / soTang;

  let svg = `<svg width="${svgW}" height="${svgH}" xmlns="http://www.w3.org/2000/svg">`;
  const x = padding, y = padding;
  const bw = 2; // border width

  if (hasHeader) {
    // Header block
    svg += `<rect x="${x}" y="${y}" width="${innerW}" height="${headerH}" fill="#1a3c6e" rx="4" ry="4"/>`;
    svg += `<text x="${x + innerW/2}" y="${y + headerH/2 + 5}" text-anchor="middle" fill="white" font-size="10" font-weight="bold" font-family="sans-serif">TỦ LẠNH/ĐÔNG</text>`;
    // Body
    svg += `<rect x="${x}" y="${y + headerH}" width="${innerW}" height="${bodyH}" fill="#e8f4fd" rx="0" ry="0"/>`;
  } else {
    // Wood shelf
    svg += `<rect x="${x}" y="${y}" width="${innerW}" height="${bodyH}" fill="#f5e6c8"/>`;
  }

  // Tầng dividers
  for (let i = 1; i < soTang; i++) {
    const ty = y + headerH + Math.round(i * tangH);
    svg += `<line x1="${x}" y1="${ty}" x2="${x + innerW}" y2="${ty}" stroke="${hasHeader ? '#a0c8e8' : '#c4a07a'}" stroke-width="1.5"/>`;
  }

  // Bottom line
  const bottomY = y + headerH + bodyH;
  svg += `<line x1="${x}" y1="${bottomY - 3}" x2="${x + innerW}" y2="${bottomY - 3}" stroke="${hasHeader ? '#1a6e9e' : '#5d3a1a'}" stroke-width="3"/>`;

  // Outer border
  if (hasHeader) {
    svg += `<rect x="${x}" y="${y}" width="${innerW}" height="${innerH}" fill="none" stroke="#1a3c6e" stroke-width="2" rx="4" ry="4"/>`;
  } else {
    svg += `<rect x="${x}" y="${y}" width="${innerW}" height="${bodyH}" fill="none" stroke="#5d3a1a" stroke-width="2"/>`;
  }

  svg += '</svg>';
  return svg;
}

// Full shelf SVG for POG canvas (pixel-precise)
function generateCanvasShelfSVG(shelf, pxW, pxH, scale) {
  const { soTang, rong, caoTang, loai } = shelf;
  const hasHeader = loai === 'tu';
  const headerPxH = hasHeader ? Math.round(pxH * 0.15) : 0;
  const bodyPxH = pxH - headerPxH;
  const tangPxH = bodyPxH / soTang;
  const shelfLineH = 6;

  let svg = `<svg width="${pxW}" height="${pxH + shelfLineH}" xmlns="http://www.w3.org/2000/svg" id="shelfSVGMain">`;

  if (hasHeader) {
    svg += `<rect x="0" y="0" width="${pxW}" height="${headerPxH}" fill="#1a3c6e"/>`;
    svg += `<text x="${pxW/2}" y="${headerPxH/2 + 6}" text-anchor="middle" fill="white" font-size="${Math.max(12, headerPxH * 0.4)}" font-weight="bold" font-family="sans-serif">TỦ LẠNH/ĐÔNG</text>`;
    svg += `<rect x="0" y="${headerPxH}" width="${pxW}" height="${bodyPxH}" fill="#e8f4fd"/>`;
  } else {
    svg += `<rect x="0" y="0" width="${pxW}" height="${bodyPxH}" fill="#f5e6c8"/>`;
  }

  // Tầng dividers
  for (let i = 1; i < soTang; i++) {
    const ty = headerPxH + Math.round(i * tangPxH);
    svg += `<line x1="0" y1="${ty}" x2="${pxW}" y2="${ty}" stroke="${hasHeader ? '#7ab8d8' : '#c4a07a'}" stroke-width="2"/>`;
  }

  // Bottom thick shelf line
  const bottomY = headerPxH + bodyPxH;
  svg += `<rect x="0" y="${bottomY}" width="${pxW}" height="${shelfLineH}" fill="${hasHeader ? '#1a3c6e' : '#5d3a1a'}"/>`;

  // Outer border
  svg += `<rect x="1" y="1" width="${pxW-2}" height="${pxH - 1}" fill="none" stroke="${hasHeader ? '#1a3c6e' : '#5d3a1a'}" stroke-width="2"/>`;

  svg += '</svg>';
  return { svg, headerPxH, bodyPxH, tangPxH, shelfLineH };
}

  let shelf;
  if (typeof shelfOrId === 'string') {
    shelf = allShelves.find(s => s.id === shelfOrId);
  } else {
    shelf = shelfOrId;
  }
  if (!shelf) return;

  currentShelf = shelf;
  selectedSlot = null;

  // Initialize POG config if new shelf or different shelf
  if (!pogConfig || pogConfig.shelfId !== shelf.id) {
    pogConfig = {
      shelfId: shelf.id,
      shelfConfig: { ...shelf },
      tầng: Array.from({ length: shelf.soTang }, (_, i) => ({ index: i, products: [] }))
    };
    currentPOGId = null;
    document.getElementById('pogNameInput').value = 'POG Mới';
    setAutosave('new');
  }

  updateShelfInfoPanel(shelf);
  renderPOGCanvas();
  markDirty();
}

    <div class="shelf-name-disp">${escHtml(shelf.ten)}</div>
    <div class="shelf-info-panel">
      ${shelf.loai === 'tu' ? '🧊 Tủ lạnh/đông' : '🗄️ Kệ đứng'}<br>
      ${shelf.soTang} tầng · ${shelf.rong}cm rộng · ${shelf.caoTang}cm/tầng
    </div>`;
}

// POG CANVAS RENDERING
// ============================================================
let SCALE; // px per cm

function computeScale(shelf) {
  const scrollEl = document.getElementById('pogCanvasScroll');
  const availW = scrollEl.clientWidth - 48;
  SCALE = Math.min((availW / shelf.rong) * zoom, 12 * zoom);
  return SCALE;
}

function setZoom(z) {
  zoom = Math.max(0.3, Math.min(3, parseFloat(z.toFixed(1))));
  document.getElementById('zoomLabel').textContent = Math.round(zoom * 100) + '%';
  if (currentShelf) renderPOGCanvas();
}

function renderPOGCanvas() {
  const container = document.getElementById('pogCanvasContainer');
  const emptyState = document.getElementById('pogEmptyState');

  if (!currentShelf || !pogConfig) {
    emptyState.style.display = 'flex';
    return;
  }
  emptyState.style.display = 'none';

  const shelf = currentShelf;
  const sc = computeScale(shelf);

  const pxW = Math.round(shelf.rong * sc);
  const hasHeader = shelf.loai === 'tu';
  const headerPxH = hasHeader ? Math.round(shelf.soTang * shelf.caoTang * sc * 0.15) : 0;
  const bodyPxH = Math.round(shelf.soTang * shelf.caoTang * sc);
  const tangPxH = Math.round(shelf.caoTang * sc);
  const shelfLineH = 6;
  const totalH = headerPxH + bodyPxH + shelfLineH;

  // Clear & render SVG
  const svgResult = generateCanvasShelfSVG(shelf, pxW, headerPxH + bodyPxH, sc);

  // Remove old shelf container if any
  const oldShelf = container.querySelector('.shelf-svg-container');
  if (oldShelf) oldShelf.remove();

  const shelfContainer = document.createElement('div');
  shelfContainer.className = 'shelf-svg-container';
  shelfContainer.style.position = 'relative';
  shelfContainer.style.display = 'inline-block';
  shelfContainer.innerHTML = svgResult.svg;

  container.insertBefore(shelfContainer, container.firstChild);

  // Render product overlays for each tầng
  // Remove old overlays
  container.querySelectorAll('.tầng-overlay').forEach(el => el.remove());

  for (let ti = 0; ti < shelf.soTang; ti++) {
    const tConfig = pogConfig.tầng[ti] || { index: ti, products: [] };
    // tầng 0 is TOP shelf, but visually we render top-to-bottom
    const offsetY = headerPxH + ti * tangPxH;

    const overlay = document.createElement('div');
    overlay.className = 'tầng-overlay';
    overlay.dataset.tang = ti;
    overlay.style.cssText = `
      position:absolute;
      left:0; top:${offsetY}px;
      width:${pxW}px; height:${tangPxH}px;
      align-items:flex-end;
      overflow:hidden;
    `;

    // Drop zone events
    overlay.addEventListener('dragover', e => { e.preventDefault(); overlay.classList.add('drag-over'); });
    overlay.addEventListener('dragleave', e => { overlay.classList.remove('drag-over'); });
    overlay.addEventListener('drop', e => {
      e.preventDefault();
      overlay.classList.remove('drag-over');
      const maSP = e.dataTransfer.getData('text/plain');
      if (maSP) {
        addProductToTang(parseInt(overlay.dataset.tang), maSP);
      }
    });
    overlay.addEventListener('click', e => {
      if (e.target === overlay && selectedLibProduct) {
        addProductToTang(parseInt(overlay.dataset.tang), selectedLibProduct.maSP);
      }
    });

    // Render products
    let xOffset = 0;
    (tConfig.products || []).forEach((prod, pi) => {
      const pxRong = Math.round(prod.rong * sc);
      const pxCao = Math.min(Math.round(prod.cao * sc), tangPxH - 4);
      const facing = prod.facing || 1;

      const slot = document.createElement('div');
      slot.className = 'product-slot';
      slot.dataset.tang = ti;
      slot.dataset.slot = pi;
      slot.style.cssText = `
        width:${pxRong}px;
        height:${tangPxH}px;
        position:absolute;
        left:${xOffset}px;
        bottom:0;
      `;

      const isSelected = selectedSlot && selectedSlot.tangIndex === ti && selectedSlot.slotIndex === pi;
      if (isSelected) slot.classList.add('selected');

      slot.addEventListener('click', e => {
        e.stopPropagation();
        selectedSlot = { tangIndex: ti, slotIndex: pi };
        renderPOGCanvas();
      });

      // Build stacked facing images
      const imgContainer = document.createElement('div');
      imgContainer.style.cssText = `
        position:absolute; bottom:0; left:0; right:0;
        display:flex; flex-direction:column; align-items:center; justify-content:flex-end;
      `;

      for (let f = 0; f < facing; f++) {
        if (prod.linkAnh) {
          const img = document.createElement('img');
          img.src = formatImgUrl(prod.linkAnh);
          img.alt = prod.tenSP || '';
          img.style.cssText = `width:${pxRong}px; height:${pxCao}px; object-fit:contain; display:block;`;
          img.onerror = () => {
            img.style.display = 'none';
          };
          imgContainer.appendChild(img);
        } else {
          const placeholder = document.createElement('div');
          placeholder.style.cssText = `
            width:${pxRong}px; height:${pxCao}px;
            background:var(--gray-200);
            display:flex; align-items:center; justify-content:center;
            font-size:${Math.max(10, Math.min(20, pxRong * 0.4))}px;
            color:var(--gray-500);
            border:1px solid var(--gray-300);
          `;
          placeholder.textContent = '📦';
          imgContainer.appendChild(placeholder);
        }
      }
      slot.appendChild(imgContainer);

      // Facing count badge
      if (facing > 1) {
        const badge = document.createElement('div');
        badge.className = 'facing-count';
        badge.textContent = `×${facing}`;
        slot.appendChild(badge);
      }

      // Controls (shown when selected)
      const controls = document.createElement('div');
      controls.className = 'slot-controls';

      const delBtn = document.createElement('button');
      delBtn.className = 'slot-btn delete';
      delBtn.textContent = '×';
      delBtn.title = 'Xoá';
      delBtn.onclick = (e) => { e.stopPropagation(); removeProductFromTang(ti, pi); };

      const faceUpBtn = document.createElement('button');
      faceUpBtn.className = 'slot-btn facing-up';
      faceUpBtn.textContent = '+';
      faceUpBtn.title = 'Thêm facing';
      faceUpBtn.onclick = (e) => { e.stopPropagation(); changeFacing(ti, pi, 1); };

      const faceDownBtn = document.createElement('button');
      faceDownBtn.className = 'slot-btn facing-down';
      faceDownBtn.textContent = '−';
      faceDownBtn.title = 'Giảm facing';
      faceDownBtn.onclick = (e) => { e.stopPropagation(); changeFacing(ti, pi, -1); };

      controls.appendChild(delBtn);
      controls.appendChild(faceUpBtn);
      controls.appendChild(faceDownBtn);
      slot.appendChild(controls);

      // Product name tooltip
      const nameEl = document.createElement('div');
      nameEl.style.cssText = `
        position:absolute; bottom:0; left:0; right:0;
        background:rgba(0,0,0,0.55); color:#fff;
        font-size:${Math.max(8, Math.min(11, pxRong * 0.12))}px;
        padding:1px 2px; text-align:center;
        white-space:nowrap; overflow:hidden; text-overflow:ellipsis;
        pointer-events:none;
      `;
      nameEl.textContent = prod.maSP || prod.tenSP || '';
      slot.appendChild(nameEl);

      overlay.appendChild(slot);
      xOffset += pxRong;
    });

    shelfContainer.appendChild(overlay);
  }

  shelfContainer.style.width = pxW + 'px';
  shelfContainer.style.height = totalH + 'px';
}

function addProductToTang(tangIndex, maSP) {
  const prod = allProducts.find(p => p.maSP === maSP);
  if (!prod) return;

  if (!pogConfig.tầng[tangIndex]) {
    pogConfig.tầng[tangIndex] = { index: tangIndex, products: [] };
  }

  const shelf = currentShelf;
  const totalWidth = pogConfig.tầng[tangIndex].products.reduce((sum, p) => sum + (p.rong || 10), 0);
  if (totalWidth + (prod.rong || 10) > shelf.rong + 1) {
    showToast('Tầng đã đầy!', 'warning');
    return;
  }

  pogConfig.tầng[tangIndex].products.push({
    maSP: prod.maSP,
    tenSP: prod.tenSP,
    linkAnh: prod.linkAnh || '',
    rong: prod.rong || 10,
    cao: prod.cao || 20,
    facing: 1
  });

  selectedSlot = { tangIndex, slotIndex: pogConfig.tầng[tangIndex].products.length - 1 };
  renderPOGCanvas();
  markDirty();
}

function removeProductFromTang(tangIndex, slotIndex) {
  if (pogConfig.tầng[tangIndex]) {
    pogConfig.tầng[tangIndex].products.splice(slotIndex, 1);
    selectedSlot = null;
    renderPOGCanvas();
    markDirty();
  }
}

function changeFacing(tangIndex, slotIndex, delta) {
  const prod = pogConfig.tầng[tangIndex]?.products[slotIndex];
  if (!prod) return;
  prod.facing = Math.max(1, Math.min(10, (prod.facing || 1) + delta));
  renderPOGCanvas();
  markDirty();
}

function onLibDragStart(event, maSP) {
  const prod = allProducts.find(p => p.maSP === maSP);
  event.dataTransfer.setData('text/plain', maSP);
  event.dataTransfer.effectAllowed = 'copy';

  const ghost = document.getElementById('dragGhost');
  const ghostImg = document.getElementById('dragGhostImg');
  const ghostName = document.getElementById('dragGhostName');

  if (prod?.linkAnh) {
    ghostImg.src = formatImgUrl(prod.linkAnh);
    ghostImg.style.display = 'block';
  } else {
    ghostImg.style.display = 'none';
  }
  ghostName.textContent = prod?.tenSP || maSP;
  ghost.classList.add('active');

  document.addEventListener('mousemove', moveDragGhost);
}

function onLibDragEnd(event) {
  document.getElementById('dragGhost').classList.remove('active');
  document.removeEventListener('mousemove', moveDragGhost);
  document.querySelectorAll('.tầng-overlay').forEach(el => el.classList.remove('drag-over'));
}

function moveDragGhost(e) {
  const ghost = document.getElementById('dragGhost');
  ghost.style.left = (e.clientX + 12) + 'px';
  ghost.style.top = (e.clientY - 20) + 'px';
}

function markDirty() {
  isDirty = true;
  setAutosave('unsaved');
  clearTimeout(autosaveTimer);
  autosaveTimer = setTimeout(() => {
    if (isDirty && currentPOGId) autoSavePOG();
  }, 5000);
}

function setAutosave(state) {
  const el = document.getElementById('autosaveIndicator');
  if (state === 'saving') {
    el.textContent = '⏳ Đang lưu...';
    el.className = 'autosave-indicator saving';
  } else if (state === 'saved') {
    el.textContent = '✅ Đã lưu';
    el.className = 'autosave-indicator saved';
  } else if (state === 'unsaved') {
    el.textContent = '● Chưa lưu';
    el.className = 'autosave-indicator';
  } else {
    el.textContent = 'POG mới';
    el.className = 'autosave-indicator';
  }
}

async function autoSavePOG() {
  if (!currentPOGId || !pogConfig) return;
  try {
    await apiPost({
      action: 'savePOG',
      id: currentPOGId,
      ten: document.getElementById('pogNameInput').value.trim() || 'POG',
      username: currentUser.username,
      shelfId: currentShelf?.id || '',
      jsonConfig: pogConfig
    });
    isDirty = false;
    setAutosave('saved');
  } catch(e) {
    console.error('Autosave failed:', e);
  }
}

async function savePOG() {
  if (!pogConfig) { showToast('Chưa có kệ nào được chọn', 'warning'); return; }
  let name = document.getElementById('pogNameInput').value.trim();
  if (!name) {
    name = prompt('Nhập tên POG:') || 'POG';
    document.getElementById('pogNameInput').value = name;
  }

  showLoading('Đang lưu POG...');
  try {
    const res = await apiPost({
      action: 'savePOG',
      id: currentPOGId || '',
      ten: name,
      username: currentUser.username,
      shelfId: currentShelf?.id || '',
      jsonConfig: pogConfig
    });
    if (res.success) {
      currentPOGId = res.id || currentPOGId;
      isDirty = false;
      setAutosave('saved');
      showToast('Đã lưu POG!', 'success');
    } else {
      showToast(res.error || 'Lỗi lưu POG', 'error');
    }
  } catch(e) {
    showToast('Lỗi: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

  openModal('modalPOGList');

  body.innerHTML = `<div class="empty-list"><div class="empty-icon">⏳</div><p>Đang tải...</p></div>`;

async function loadPOG(id) {
  closeModal('modalPOGList');
  showLoading('Đang tải POG...');
  try {
    const data = await apiGet({ action: 'getPOGDetail', id });
    if (!data.success) { showToast(data.error || 'Không tìm thấy POG', 'error'); return; }

    currentPOGId = data.id;
    document.getElementById('pogNameInput').value = data.ten || 'POG';
    pogConfig = data.jsonConfig || {};

    // Load shelf
    const shelfId = data.shelfId || pogConfig.shelfId;
    let shelf = allShelves.find(s => s.id === shelfId);
    if (!shelf && pogConfig.shelfConfig) {
      shelf = pogConfig.shelfConfig;
    }
    if (shelf) {
      currentShelf = shelf;
      // Ensure tầng config matches shelf
      const existingTangs = pogConfig.tầng || [];
      pogConfig.tầng = Array.from({ length: shelf.soTang }, (_, i) => {
        return existingTangs.find(t => t.index === i) || { index: i, products: [] };
      });
      updateShelfInfoPanel(shelf);
    }

    renderPOGCanvas();
    setAutosave('saved');
    isDirty = false;
    showToast('Đã tải POG!', 'success');
  } catch(e) {
    showToast('Lỗi: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

// EXPORT PNG
// ============================================================
async function exportPNG() {
  if (!currentShelf || !pogConfig) {
    showToast('Chưa có kệ để xuất', 'warning');
    return;
  }

  showLoading('Đang xuất PNG...');

  try {
    // Use canvas-based export for reliability
    const A4_W = 2480;
    const shelf = currentShelf;
    const hasHeader = shelf.loai === 'tu';

    // Calculate scale to fit A4 landscape
    const margin = 80;
    const exportScale = (A4_W - margin * 2) / shelf.rong;

    const pxW = Math.round(shelf.rong * exportScale);
    const headerPxH = hasHeader ? Math.round(shelf.soTang * shelf.caoTang * exportScale * 0.15) : 0;
    const bodyPxH = Math.round(shelf.soTang * shelf.caoTang * exportScale);
    const tangPxH = Math.round(shelf.caoTang * exportScale);
    const shelfLineH = 12;
    const totalH = headerPxH + bodyPxH + shelfLineH + margin * 2;
    const totalW = A4_W;

    const canvas = document.createElement('canvas');
    canvas.width = totalW;
    canvas.height = totalH;
    const ctx = canvas.getContext('2d');

    // White background
    ctx.fillStyle = '#ffffff';
    ctx.fillRect(0, 0, totalW, totalH);

    const startX = margin;
    const startY = margin;

    // Draw header if fridge
    if (hasHeader) {
      ctx.fillStyle = '#1a3c6e';
      ctx.fillRect(startX, startY, pxW, headerPxH);
      ctx.fillStyle = '#ffffff';
      ctx.font = `bold ${Math.max(16, headerPxH * 0.4)}px sans-serif`;
      ctx.textAlign = 'center';
      ctx.fillText('TỦ LẠNH/ĐÔNG', startX + pxW / 2, startY + headerPxH / 2 + 8);
    }

    // Draw shelf body
    ctx.fillStyle = hasHeader ? '#e8f4fd' : '#f5e6c8';
    ctx.fillRect(startX, startY + headerPxH, pxW, bodyPxH);

    // Draw tầng dividers
    ctx.strokeStyle = hasHeader ? '#7ab8d8' : '#c4a07a';
    ctx.lineWidth = 2;
    for (let i = 1; i < shelf.soTang; i++) {
      const ty = startY + headerPxH + Math.round(i * tangPxH);
      ctx.beginPath();
      ctx.moveTo(startX, ty);
      ctx.lineTo(startX + pxW, ty);
      ctx.stroke();
    }

    // Bottom shelf line
    ctx.fillStyle = hasHeader ? '#1a3c6e' : '#5d3a1a';
    ctx.fillRect(startX, startY + headerPxH + bodyPxH, pxW, shelfLineH);

    // Outer border
    ctx.strokeStyle = hasHeader ? '#1a3c6e' : '#5d3a1a';
    ctx.lineWidth = 4;
    ctx.strokeRect(startX, startY, pxW, headerPxH + bodyPxH);

    // Draw products (load images async)
    const imagePromises = [];

    for (let ti = 0; ti < shelf.soTang; ti++) {
      const tConfig = pogConfig.tầng[ti] || { products: [] };
      const tangStartY = startY + headerPxH + ti * tangPxH;

      let xOff = 0;
      for (const prod of tConfig.products) {
        const pxRong = Math.round(prod.rong * exportScale);
        const pxCao = Math.min(Math.round(prod.cao * exportScale), tangPxH - 4);
        const facing = prod.facing || 1;

        for (let f = 0; f < facing; f++) {
          const imgY = tangStartY + tangPxH - (f + 1) * pxCao;
          const imgX = startX + xOff;

          if (prod.linkAnh) {
            const p = new Promise(resolve => {
              const img = new Image();
              img.crossOrigin = 'anonymous';
              img.onload = () => {
                ctx.drawImage(img, imgX, imgY, pxRong, pxCao);
                resolve();
              };
              img.onerror = () => {
                // Placeholder
                ctx.fillStyle = '#e0e0e0';
                ctx.fillRect(imgX, imgY, pxRong, pxCao);
                ctx.fillStyle = '#999';
                ctx.font = '14px sans-serif';
                ctx.textAlign = 'center';
                ctx.fillText('📦', imgX + pxRong/2, imgY + pxCao/2 + 5);
                resolve();
              };
              img.src = formatImgUrl(prod.linkAnh);
            });
            imagePromises.push(p);
          } else {
            ctx.fillStyle = '#e9ecef';
            ctx.fillRect(imgX, imgY, pxRong, pxCao);
            ctx.fillStyle = '#adb5bd';
            ctx.font = `${Math.max(12, pxRong * 0.3)}px sans-serif`;
            ctx.textAlign = 'center';
            ctx.fillText('📦', imgX + pxRong/2, imgY + pxCao/2 + 5);
          }
        }

        // Product label
        const labelY = tangStartY + tangPxH - 2;
        ctx.fillStyle = 'rgba(0,0,0,0.6)';
        ctx.fillRect(startX + xOff, tangStartY + tangPxH - 18, pxRong, 18);
        ctx.fillStyle = '#ffffff';
        ctx.font = `${Math.max(9, Math.min(13, pxRong * 0.12))}px sans-serif`;
        ctx.textAlign = 'center';
        ctx.fillText(prod.maSP || prod.tenSP || '', startX + xOff + pxRong/2, labelY - 4);

        xOff += pxRong;
      }
    }

    // Title
    const pogName = document.getElementById('pogNameInput').value.trim() || 'POG';
    ctx.fillStyle = '#1a3c6e';
    ctx.font = `bold 36px sans-serif`;
    ctx.textAlign = 'left';
    ctx.fillText(`📐 ${pogName}`, margin, margin - 20);
    ctx.font = '22px sans-serif';
    ctx.fillStyle = '#6c757d';
    ctx.textAlign = 'right';
    ctx.fillText(new Date().toLocaleDateString('vi-VN'), totalW - margin, margin - 20);

    // Wait for all images
    await Promise.all(imagePromises);

    // Re-draw borders on top of images
    ctx.strokeStyle = hasHeader ? '#1a3c6e' : '#5d3a1a';
    ctx.lineWidth = 4;
    ctx.strokeRect(startX, startY, pxW, headerPxH + bodyPxH);

    // Tầng dividers on top
    ctx.strokeStyle = hasHeader ? '#7ab8d8' : '#c4a07a';
    ctx.lineWidth = 2;
    for (let i = 1; i < shelf.soTang; i++) {
      const ty = startY + headerPxH + Math.round(i * tangPxH);
      ctx.beginPath();
      ctx.moveTo(startX, ty);
      ctx.lineTo(startX + pxW, ty);
      ctx.stroke();
    }

    // Download
    const dateStr = new Date().toISOString().slice(0,10);
    const filename = `POG_${pogName.replace(/[^a-zA-Z0-9_\u00C0-\u024F\u1E00-\u1EFF]/g, '_')}_${dateStr}.png`;
    const link = document.createElement('a');
    link.download = filename;
    link.href = canvas.toDataURL('image/png');
    link.click();

    // Log export
    if (currentPOGId) {
      apiPost({ action: 'exportLog', pogId: currentPOGId, username: currentUser.username }).catch(() => {});
    }

    showToast('Đã xuất PNG!', 'success');
  } catch(e) {
    console.error(e);
    showToast('Lỗi xuất PNG: ' + e.message, 'error');
  } finally {
    hideLoading();
  }
}

function escHtml(str) {
  if (!str) return '';
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function formatImgUrl(url) {
  if (!url) return '';
  // Convert Google Drive share links to direct view links
  if (url.includes('drive.google.com/file/d/')) {
    const m = url.match(/\/file\/d\/([^\/\?]+)/);
    if (m) return `https://drive.google.com/uc?export=view&id=${m[1]}`;
  }
  if (url.includes('drive.google.com/open?id=')) {
    const m = url.match(/[?&]id=([^&]+)/);
    if (m) return `https://drive.google.com/uc?export=view&id=${m[1]}`;
  }
  return url;
}

function formatDate(isoStr) {
  if (!isoStr) return '—';
  try {
    const d = new Date(isoStr);
    return d.toLocaleDateString('vi-VN');
  } catch(e) {
    return isoStr.slice(0, 10);
  }
}
